10.3 Package

おすすめは、「sf」です。 また、今回は、QGISのアルゴリズムで作成する方法も解説します。 こちらは「qgisprocess」です。

library(qgisprocess) #QGIS Algorithm
library(dplyr)
library(sf) #GIS用
library(leaflet) #Map

10.4 入力データを可視化

ノースカロライナを可視化します。

#Leafletに出力するため、WGS84に変更する
input_disp = input %>% st_transform(crs=4326)

#Leafletに出力
leaflet() %>%
  addTiles() %>%
  addPolygons(data = input_disp, color="blue")

10.5 sfで5kmのBufferを作成

測地系（CRS）を設定して、st_union()で合成し、 st_buffer()で、5000mのバッファを作成します。 赤色の線で、Bufferしたポリゴンを表します。

#meter bufferを作成するため、ノースカロライナのEPSGに変換(st_transform)する。
#各州を結合(st_union)して、sfのBuffer作成する。
sf_ret = input %>% st_transform(crs=26717) %>%
  st_union() %>% st_as_sf() %>% 
  st_buffer(dist = 5000)

#表示用にWGS84に直す
b_sf = sf_ret %>% st_transform(crs=4326)

#Leafletで表示
leaflet() %>%
  addTiles() %>%
  addPolygons(data = input_disp, color="blue") %>%
  addPolylines(data = b_sf, weight=10,color="red") %>%
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))

10.7 QGISで1kmのBufferを作成

測地系（CRS）を設定して、st_union()で合成し、 QGIS Algorithm 「native:buffer」で、1000mのバッファを作成します。 赤色の線で、Bufferしたポリゴンを表します。

#meter bufferのため、ノースカロライナのEPSGに変換する。
input_a = input %>% 
  st_union() %>% st_as_sf() %>% 
  st_transform(crs=26717)
input_a
## Simple feature collection with 1 feature and 0 fields
## Geometry type: MULTIPOLYGON
## Dimension:     XY
## Bounding box:  xmin: 196586.7 ymin: 3751514 xmax: 1002247 ymax: 4057630
## Projected CRS: NAD27 / UTM zone 17N
##                                x
## 1 MULTIPOLYGON (((949976.3 40...
#QGISのBuffer処理(1000 meter)
q_alg = "native:buffer"
q_alg_ret = qgis_run_algorithm(q_alg,
                               INPUT = input_a,
                               DISTANCE = 1000
)

#各州を結合して、表示用にWGS84に直す
a_sf = st_as_sf(q_alg_ret) %>% st_transform(crs=4326)

#Leafletで表示
leaflet() %>%
  addTiles() %>%
  addPolygons(data = input_disp, color="blue") %>%
  addPolylines(data = b_sf, weight=10,color="red") %>%
  addPolylines(data = a_sf, weight=5,color="green") %>%
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))
