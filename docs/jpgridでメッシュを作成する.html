12 jpgridでメッシュを作成する

12.1 Package

日本のGIS分析をするときにメッシュを使いますよね。 Rで日本のメッシュなら、「jpgrid」を使います。（新しい！！）

library(leaflet)
library(sf)
library(jpgrid)
library(dplyr)

12.2 札幌市のデータ

国土数値情報ダウンロードサイトにあるシェープファイルを読み込みます。

# 国土数値情報 行政区域データの読み込み
read_data = read_sf("N03-20200101_GML/N03-20_200101.shp",
                    options = "ENCODING=CP932")
#札幌市のデータ（WGS84）
a_sf = subset(read_data, read_data$N03_003 == "札幌市") %>% st_union() %>% st_transform(4326)

#output
leaflet() %>%
  addTiles() %>%
  addPolygons(data=a_sf) %>% 
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))

12.3 ポリゴンからメッシュを作成

札幌市のポリゴンから、10kmメッシュを作成します。

#Create 10km Mesh
grid_10km = a_sf %>% geometry_to_grid("10km") %>% do.call("c", .)
m_sf_10km = grid_10km %>% grid_as_sf(crs = 4326)

#output
leaflet() %>%
  addTiles() %>%
  addPolygons(data=a_sf) %>% 
  addPolygons(data=m_sf_10km, color="grey") %>% 
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))

12.4 10kmメッシュを1kmメッシュに分解

作成した10kmメッシュを分解して、1kmメッシュにします。

#10 km Mesh to 1 km Mesh
grid_1km = grid_subdivide(grid_10km, "1km") %>% do.call("c", .)
m_sf_1km = grid_1km %>% grid_as_sf(crs = 4326)

leaflet() %>%
  addTiles() %>%
  addPolygons(data=a_sf) %>% 
  addPolygons(data=m_sf_1km, color="grey") %>% 
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))

12.5 緯度経度からメッシュ作成

さくっと作成できます。

grid_100m = coords_to_grid(139.7008,35.68906,"100m")
m_sf_100m = grid_100m %>% grid_as_sf(crs = 4326)

leaflet() %>%
  addTiles() %>%
  addPolygons(data=m_sf_100m) %>% 
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))

12.6 隣接メッシュ作成（ノイマン近傍）

メッシュに隣接するメッシュを作成できます。すごごご。

grid_100m_n = grid_100m %>% grid_neighborhood(
  n = c(0:2),
  type = "von_neumann",
  simplify = FALSE)
#grid_100m_n
m_sf_100m_n = grid_100m_n[[1]]$grid_neighborhood %>% grid_as_sf(crs = 4326)
#m_sf_100m_n
m_sf_100m_nm = merge(m_sf_100m_n,grid_100m_n[1],
                     by.x = c("grid"), 
                     by.y = c("grid_neighborhood"),
                     all.x = TRUE)
#m_sf_100m_nm
#class(m_sf_100m_nm$n)
pal = colorNumeric("viridis", domain=c(0:2))

leaflet() %>%
  addTiles() %>%
  addPolygons(data=m_sf_100m_nm, 
              fillOpacity = 0.7,
              fillColor = ~pal(n)
              ) %>% 
  addLegend(data=m_sf_100m_nm,
            position = "bottomright",
            values = ~n, 
            pal = pal,
            opacity = 1.0,
            title = "n" ,
            labFormat = labelFormat()) %>% 
  addScaleBar(position = "bottomleft",scaleBarOptions(imperial = FALSE))
